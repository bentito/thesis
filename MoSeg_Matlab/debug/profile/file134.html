                485     if display > 2 % 'iter'
                486         disp(sprintf(dispfmt,iter,1,length(moved),totsumD));
                487     end
                488     if iter >= maxit
                489         break;
                490     end
                491 
                492     % Determine closest cluster for each point and reassign points to clusters
                493     previdx = idx;
                494     prevtotsumD = totsumD;
                495     [d, nidx] = min(D, [], 2);
                496 
                497     % Determine which points moved
                498     moved = find(nidx ~= previdx);
                499     if ~isempty(moved)
                500         % Resolve ties in favor of not moving
                501         moved = moved(D((previdx(moved)-1)*n + moved) > d(moved));
                502     end
                503     if isempty(moved)
                504         converged = true;
                505         break;
                506     end
                507     idx(moved) = nidx(moved);
                508 
                509     % Find clusters that gained or lost members
                510     changed = unique([idx(moved); previdx(moved)])';
                511 
                512 end % phase one
                513 
                514 end % nested function

                516 %------------------------------------------------------------------
                517 
                518 function converged = onlineUpdate
                519 
                520 % Initialize some cluster information prior to phase two
                521 switch distance
                522 case 'cityblock'
                523     Xmid = zeros([k,p,2]);
                524     for i = 1:k
                525         if m(i) > 0
                526             % Separate out sorted coords for points in i'th cluster,
                527             % and save values above and below median, component-wise
                528             Xsorted = sort(X(idx==i,:),1);
                529             nn = floor(.5*m(i));
                530             if mod(m(i),2) == 0
                531                 Xmid(i,:,1:2) = Xsorted([nn, nn+1],:)';
                532             elseif m(i) > 1
                533                 Xmid(i,:,1:2) = Xsorted([nn, nn+2],:)';
                534             else
                535                 Xmid(i,:,1:2) = Xsorted([1, 1],:)';
                536             end
                537         end
                538     end
                539 case 'hamming'
                540     Xsum = zeros(k,p);
                541     for i = 1:k
                542         if m(i) > 0
                543             % Sum coords for points in i'th cluster, component-wise
                544             Xsum(i,:) = sum(X(idx==i,:), 1);
                545         end
                546     end
                547 end
                548 
                549 %
                550 % Begin phase two:  single reassignments
                551 %
                552 changed = find(m' > 0);
                553 lastmoved = 0;
                554 nummoved = 0;
                555 iter1 = iter;
                556 converged = false;
                557 while iter < maxit
                558     % Calculate distances to each cluster from each point, and the
                559     % potential change in total sum of errors for adding or removing
                560     % each point from each cluster.  Clusters that have not changed
                561     % membership need not be updated.
                562     %
                563     % Singleton clusters are a special case for the sum of dists
                564     % calculation.  Removing their only point is never best, so the
                565     % reassignment criterion had better guarantee that a singleton
                566     % point will stay in its own cluster.  Happily, we get
                567     % Del(i,idx(i)) == 0 automatically for them.
                568     switch distance
                569     case 'sqeuclidean'
                570         for i = changed
                571             mbrs = (idx == i);
                572             sgn = 1 - 2*mbrs; % -1 for members, 1 for nonmembers
                573             if m(i) == 1
                574                 sgn(mbrs) = 0; % prevent divide-by-zero for singleton mbrs
                575             end
                576             Del(:,i) = (m(i) ./ (m(i) + sgn)) .* sum((X - C(repmat(i,n,1),:)).^2, 2);
                577         end
                578     case 'cityblock'
                579         for i = changed
                580             if mod(m(i),2) == 0 % this will never catch singleton clusters
                581                 ldist = Xmid(repmat(i,n,1),:,1) - X;
                582                 rdist = X - Xmid(repmat(i,n,1),:,2);
                583                 mbrs = (idx == i);
                584                 sgn = repmat(1-2*mbrs, 1, p); % -1 for members, 1 for nonmembers
                585                 Del(:,i) = sum(max(0, max(sgn.*rdist, sgn.*ldist)), 2);
                586             else
                587                 Del(:,i) = sum(abs(X - C(repmat(i,n,1),:)), 2);
                588             end
                589         end
                590     case {'cosine','correlation'}
                591         % The points are normalized, centroids are not, so normalize them
                592         normC = sqrt(sum(C.^2, 2));
                593         if any(normC < eps(class(normC))) % small relative to unit-length data points
                594             error('stats:kmeans:ZeroCentroid', ...
                595                   'Zero cluster centroid created at iteration %d%s.',iter,repsMsg(rep,reps));
                596         end
                597         % This can be done without a loop, but the loop saves memory allocations
                598         for i = changed
                599             XCi = X * C(i,:)';
                600             mbrs = (idx == i);
                601             sgn = 1 - 2*mbrs; % -1 for members, 1 for nonmembers
                602             Del(:,i) = 1 + sgn .*...
                603                   (m(i).*normC(i) - sqrt((m(i).*normC(i)).^2 + 2.*sgn.*m(i).*XCi + 1));
                604         end
                605     case 'hamming'
                606         for i = changed
                607             if mod(m(i),2) == 0 % this will never catch singleton clusters
                608                 % coords with an unequal number of 0s and 1s have a
                609                 % different contribution than coords with an equal
                610                 % number
                611                 unequal01 = find(2*Xsum(i,:) ~= m(i));
                612                 numequal01 = p - length(unequal01);
                613                 mbrs = (idx == i);
                614                 Di = abs(X(:,unequal01) - C(repmat(i,n,1),unequal01));
                615                 Del(:,i) = (sum(Di, 2) + mbrs*numequal01) / p;
                616             else
                617                 Del(:,i) = sum(abs(X - C(repmat(i,n,1),:)), 2) / p;
                618             end
                619         end
                620     end
                621 
                622     % Determine best possible move, if any, for each point.  Next we
                623     % will pick one from those that actually did move.
                624     previdx = idx;
                625     prevtotsumD = totsumD;
                626     [minDel, nidx] = min(Del, [], 2);
                627     moved = find(previdx ~= nidx);
                628     if ~isempty(moved)
                629         % Resolve ties in favor of not moving
                630         moved = moved(Del((previdx(moved)-1)*n + moved) > minDel(moved));
                631     end
                632     if isempty(moved)
                633         % Count an iteration if phase 2 did nothing at all, or if we're
                634         % in the middle of a pass through all the points
                635         if (iter == iter1) || nummoved > 0
                636             iter = iter + 1;
                637             if display > 2 % 'iter'
                638                 disp(sprintf(dispfmt,iter,2,nummoved,totsumD));
                639             end
                640         end
                641         converged = true;
                642         break;
                643     end
                644 
                645     % Pick the next move in cyclic order
                646     moved = mod(min(mod(moved - lastmoved - 1, n) + lastmoved), n) + 1;
                647 
                648     % If we've gone once through all the points, that's an iteration
                649     if moved <= lastmoved
                650         iter = iter + 1;
                651         if display > 2 % 'iter'
                652             disp(sprintf(dispfmt,iter,2,nummoved,totsumD));
                653         end
                654         if iter >= maxit, break; end
                655         nummoved = 0;
                656     end
                657     nummoved = nummoved + 1;
                658     lastmoved = moved;
                659 
                660     oidx = idx(moved);
                661     nidx = nidx(moved);
                662     totsumD = totsumD + Del(moved,nidx) - Del(moved,oidx);
                663 
                664     % Update the cluster index vector, and the old and new cluster
                665     % counts and centroids
                666     idx(moved) = nidx;
                667     m(nidx) = m(nidx) + 1;
                668     m(oidx) = m(oidx) - 1;
                669     switch distance
                670     case 'sqeuclidean'
                671         C(nidx,:) = C(nidx,:) + (X(moved,:) - C(nidx,:)) / m(nidx);
                672         C(oidx,:) = C(oidx,:) - (X(moved,:) - C(oidx,:)) / m(oidx);
                673     case 'cityblock'
                674         for i = [oidx nidx]
                675             % Separate out sorted coords for points in each cluster.
                676             % New centroid is the coord median, save values above and
                677             % below median.  All done component-wise.
                678             Xsorted = sort(X(idx==i,:),1);
                679             nn = floor(.5*m(i));
                680             if mod(m(i),2) == 0
                681                 C(i,:) = .5 * (Xsorted(nn,:) + Xsorted(nn+1,:));
                682                 Xmid(i,:,1:2) = Xsorted([nn, nn+1],:)';
                683             else
                684                 C(i,:) = Xsorted(nn+1,:);
                685                 if m(i) > 1
                686                     Xmid(i,:,1:2) = Xsorted([nn, nn+2],:)';
                687                 else
                688                     Xmid(i,:,1:2) = Xsorted([1, 1],:)';
                689                 end
                690             end
                691         end
                692     case {'cosine','correlation'}
                693         C(nidx,:) = C(nidx,:) + (X(moved,:) - C(nidx,:)) / m(nidx);
                694         C(oidx,:) = C(oidx,:) - (X(moved,:) - C(oidx,:)) / m(oidx);
                695     case 'hamming'
                696         % Update summed coords for points in each cluster.  New
                697         % centroid is the coord median.  All done component-wise.
                698         Xsum(nidx,:) = Xsum(nidx,:) + X(moved,:);
                699         Xsum(oidx,:) = Xsum(oidx,:) - X(moved,:);
                700         C(nidx,:) = .5*sign(2*Xsum(nidx,:) - m(nidx)) + .5;
                701         C(oidx,:) = .5*sign(2*Xsum(oidx,:) - m(oidx)) + .5;
                702     end
                703     changed = sort([oidx nidx]);
                704 end % phase two
                705 
                706 end % nested function
                707 
             1  708 end % main function
